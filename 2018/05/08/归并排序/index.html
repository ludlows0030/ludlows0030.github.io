归并排序

归并排序辅助数组只有原数组一半大的归并排序

归并排序的思想是通过将两个有序的序列合并成一个有序的序列。
当left[i]>right[j] origin[k] = right[j]; j++;
当一边取完就从另一边取
将一个无序的数组通过迭代的方法分至最小，即由一个元素构成的序列进行排序即自顶而下的归并排序

import edu.princeton.cs.algs4.StdOut;
public class Merge{
  private static Comparable[] aux;

  @SuppressWarnings("unchecked")
  private static boolean less(Comparable v, Comparable w){
    return v.compareTo(w)<0;
  }

  public static void sort(Comparable[] a){
    aux = new Comparable[a.length];
    sort(a,0,a.length-1);
  }

  private static void sort(Comparable[] a,int lo ,int hi){
    if(hi<=lo)return;
    int mid = lo+(hi-lo)/2;
    sort(a,lo,mid);
    sort(a,mid+1,hi);
    merge(a,lo,mid,hi);
  }

  public static void merge(Comparable[] a,int lo,int mid, int hi){
    int i = lo,j = mid+1;

    for(int k = lo;k<=hi;k++){
        aux[k] = a[k];
    }

    for(int k = lo;k<=hi;k++){
        if(i>mid) a[k] = a[j++];
        else if(j>hi) a[k] = aux[i++];
        else if(less(aux[i],aux[j])) a[k] = aux[i++];
        else a[k] = aux[j++];
    }   
  }

  public static void main(String[] args){
    Comparable[] a = {2,4,6,8,0,1,3,7};
    Merge m = new Merge();
    m.sort(a);
    for(Comparable c :a){
        StdOut.println(c);
    }
  }
}

当用只有原数组一半大的辅助数组时，只将原数组的左半复制到辅助数组中

import edu.princeton.cs.algs4.StdOut;
public class Merge{
  private static Comparable[] aux;

  @SuppressWarnings("unchecked")
  private static boolean less(Comparable v, Comparable w){
    return v.compareTo(w)<0;
  }

  public static void sort(Comparable[] a){
    aux = new Comparable[a.length/2];
    sort(a,0,a.length-1);
  }

  private static void sort(Comparable[] a,int lo ,int hi){
    if(hi<=lo)return;
    int mid = lo+(hi-lo)/2;
    merge(a,lo,mid,hi);
  }

  public static void merge(Comparable[] a,int lo,int mid, int hi){
    int i = lo,j = mid+1;

    for(int k = lo;k<=mid;k++){
        aux[k] = a[k];
    }

    for(int k = lo;k<=hi;k++){
        if(i>mid) a[k] = a[j++];
        else if(j>hi) a[k] = aux[i++];
        else if(less(aux[i],a[j])) a[k] = aux[i++];
        else a[k] = a[j++];
    }   
  }

  public static void main(String[] args){
    Comparable[] a = {2,4,6,8,0,1,3,7};
    Merge m = new Merge();
    m.sort(a);
    for(Comparable c :a){
        StdOut.println(c);
    }
  }
}
