快速排序

关于快速排序需要了解的

  • 快速排序为一种原地排序
  • 快速排序是一种分治的排序算法。将一个数组分成两个子数组，将两部分独立地进行排序
  • 快速排序与归并排序的不同
    • 归并排序将数组分成两个子数组分别排序，再将有序的子数组合并使得整个数组变得有序
    • 快速排序将数组分成两个子数组（以一个数k为分界，比k小的在左边，大的在右边），当子数组有序时，整个数组有序
  • 快速排序的切分策略
    • 先随意的取a[lo]为切分元素
    • 然后从数组左端开始扫描直到找到一个大于或等于它的元素
    • 再从数组右端扫描直到找到一个小于或等于它的元素
    • 将两者交换
    • 如此继续，扫描完毕时（两个扫描指针i,j相遇），将a[lo]与a[j]交换，将切分元素放到正确的位置
    • 保证a[j]左侧的元素都小于a[j]，右侧的元素都大于a[j]

  @SuppressWarnings("unchecked")
  private static boolean less(Comparable v, Comparable w){
      return v.compareTo(w)<0;
  }

      private static void exch(Comparable[] a,int i ,int j){
      Comparable t = a[i];
      a[i] = a[j];
      a[j] = t;
  }

  public static void sort(Comparable[] a){
      StdRandom.shuffle(a);
      sort(a,0,a.length-1);
  }

  private static void sort(Comparable[] a,int lo ,int hi){
      if(hi<=lo) return;
      int j = partition(a,lo,hi);
      sort(a,lo,j-1);
      sort(a,j+1,hi);
  }

  private static partition(Comparable[] a,int lo, int hi)
  {
      int i = lo ,j = hi+1;
      Comparable v = a[lo];
      while(true)
      {
          while(less(a[++i],v)) if(i == hi) break;
          while(less(v,a[--j])) if(j == lo) break;
          exch(a,i,j);
      }
      exch(a,lo,j);//j放到正确的位置
      return j;
  }

  /*
   * 三向切分快速排序
   * private static sort(Comparable[] a, int lo, int hi)

归并排序辅助数组只有原数组一半大的归并排序

归并排序的思想是通过将两个有序的序列合并成一个有序的序列。
当left[i]>right[j] origin[k] = right[j]; j++;
当一边取完就从另一边取
将一个无序的数组通过迭代的方法分至最小，即由一个元素构成的序列进行排序即自顶而下的归并排序

import edu.princeton.cs.algs4.StdOut;
public class Merge{
  private static Comparable[] aux;

  @SuppressWarnings("unchecked")
  private static boolean less(Comparable v, Comparable w){
    return v.compareTo(w)<0;
  }

  public static void sort(Comparable[] a){
    aux = new Comparable[a.length];
    sort(a,0,a.length-1);
  }

  private static void sort(Comparable[] a,int lo ,int hi){
    if(hi<=lo)return;
    int mid = lo+(hi-lo)/2;
    sort(a,lo,mid);
    sort(a,mid+1,hi);
    merge(a,lo,mid,hi);
  }

  public static void merge(Comparable[] a,int lo,int mid, int hi){
    int i = lo,j = mid+1;

    for(int k = lo;k<=hi;k++){
        aux[k] = a[k];
    }

    for(int k = lo;k<=hi;k++){
        if(i>mid) a[k] = a[j++];
        else if(j>hi) a[k] = aux[i++];
        else if(less(aux[i],aux[j])) a[k] = aux[i++];
        else a[k] = aux[j++];
    }   
  }

  public static void main(String[] args){
    Comparable[] a = {2,4,6,8,0,1,3,7};
    Merge m = new Merge();
    m.sort(a);
    for(Comparable c :a){
        StdOut.println(c);
    }
  }
}

希尔排序

以插入排序为基础，将每一个对象与其前h个对象进行比较（h为一个递增序列通常为3h+1,比较进行一轮后h/3），当h=1时相当于进行插入排序。原因是插入排序对h有序（对象间隔h个有序）的队列排序时速度较快。

public class Shell{

        int N = a.length;

        while(h>=1){
            for(int i = h;i<N;i++){
                for(int j = i;j>0&&less(a[j],a[j-h]);j-=h){
                    exch(a,j,j-h);
                }
            }
            h = h/3;
        }

import edu.princeton.cs.algs4.LinkedStack;
import java.util.Iterator;
import edu.princeton.cs.algs4.StdOut;
public class Deque<Item> implements Iterable<Item>{
  private LinkedStack<Item> headStack;
  private LinkedStack<Item> endStack;  
  //维护两个栈一个负责头部的加入和移除一个负责尾部的

  public Deque(){
      headStack = new LinkedStack<Item>();
      endStack = new LinkedStack<Item>();
  }

  public boolean isEmpty(){
      return headStack.isEmpty()&&endStack.isEmpty();
      //如果两个栈都为空的话则整个栈为空
  }

  public int size(){
      return headStack.size()+endStack.size();
      //两个栈的size相加等于整个栈
  }

  public void addFirst(Item item){
      if(item != null){
          headStack.push(item);
      }
      else{
          throw new java.lang.IllegalArgumentException();
      }
      //在栈的最前端加入元素等于在头部栈的栈顶加入元素，头部栈的栈顶是整个栈的最前端，尾部栈的栈顶是整个栈的最末端
  }

  public void addLast(Item item){
      if(item != null){
           endStack.push(item);
      }
      else{
          throw new java.lang.IllegalArgumentException();
      }       
  }

  public Item removeFirst(){
      if(!isEmpty()){                    
        //如果头部栈为空的话，要移除第一个元素，需要将尾部栈的元素倒序复制到头部栈
          if(headStack.isEmpty()){
              while(!endStack.isEmpty()){
                  headStack.push(endStack.pop());
              }
          }
          return headStack.pop();
      }
      else throw new java.util.NoSuchElementException();

  }

  public Item removeLast(){
      if(!isEmpty()){         
        //如果尾部栈为空的话，要移除最后一个元素，需要将头部栈的元素倒序复制到尾部栈           
          if(endStack.isEmpty()){
              while(!headStack.isEmpty()){
                  endStack.push(headStack.pop());
              }
          }
          return headStack.pop();
      }
      else throw new java.util.NoSuchElementException();        
  }

  public Iterator<Item> iterator(){
      return new DequeIterator();
  }

  private class DequeIterator implements Iterator<Item>{
      private int i = size();

      Iterator<Item> head = headStack.iterator();
      Iterator<Item> end = endStack.iterator();

      public boolean hasNext(){return i>0;}
      public Item next(){
          if(!hasNext()){throw new java.util.NoSuchElementException();}
          if(!headStack.isEmpty()){
              i--;
              return head.next();
          }
          else if (headStack.isEmpty()){
              while(!endStack.isEmpty()){
                  headStack.push(endStack.pop());
              }
              i--;
              return head.next();
          }  
          else throw new java.util.NoSuchElementException();
      }
      public void remove(){
          throw new java.lang.UnsupportedOperationException();
      }
  }
  //测试函数
  public static void main(String[] args){
      Deque<Integer> dq = new Deque<Integer>();
      dq.addFirst(1);
      dq.addLast(2);
      dq.addFirst(4);
      StdOut.println(dq.removeFirst());
      StdOut.println(dq.removeFirst());
      StdOut.println(dq.removeFirst());
      StdOut.println(dq.size());
  }
}

import java.util.Iterator;
import edu.princeton.cs.algs4.StdRandom;
import edu.princeton.cs.algs4.StdOut;
public class RandomizedQueue<T> implements Iterable<T>{

  private T[] s;//用数组实现队列
  private int N;//当前数组中元素个数

  @SuppressWarnings("unchecked")
  public RandomizedQueue(){
    s = (T[])new Comparable<?>[1];
    N = 0;
  }

  @SuppressWarnings("unchecked")
  private void resize(int capacity){
    //数组大小重置为capacity
    T[] copy = (T[])new Comparable<?>[capacity];
    for(int i = 0;i<N;i++){
        copy[i]=s[i];
    }
    s = copy;
  }    

  public boolean isEmpty(){
    return N == 0;
  }

  public int size(){
    return N;
  }

  public void enqueue(T item){
    if(item ==null){
        throw new java.lang.IllegalArgumentException();
    }
    if(N == s.length){
      //如果数组满了，更改数组大小为当前两倍
        resize(2*s.length);
    }
    s[N++] = item;      
  }

  public T dequeue(){
    if(isEmpty()){
        throw new java.util.NoSuchElementException();
    }        
    int index = StdRandom.uniform(N);
    T temp = s[index];
    s[index] = s[N-1];
    s[N-1] = temp;
    //交换随机索引和最后一个索引对应的值
    T item = s[--N];
    s[N] = null;
    if(N>0&&N<s.length/4){
      //如果数组不满1/4，更改数组大小为当前的1/2
        resize(s.length/2);
    }
    return item;
  }

  public T sample(){
    int index = StdRandom.uniform(N);
    return s[index];
  }

  public Iterator<T> iterator(){
    return new RandomIterator();
  }

  private class RandomIterator implements Iterator<T>{
    private int num =N;     

    public boolean hasNext(){
        return N==0;
    }

    public T next(){
        if(num>0){
            int index = StdRandom.uniform(num--);
            T temp = s[index];
            s[index] = s[num];
            s[num]= temp;
            return temp;      
        }
        else{
            throw new java.util.NoSuchElementException();
        }
    }

    public void remove(){
        throw new java.lang.UnsupportedOperationException();
    }
  }

  public static void main(String[] args){
    RandomizedQueue<Integer> rq = new RandomizedQueue<Integer>();
    rq.enqueue(1);
    rq.enqueue(45);
    rq.enqueue(23);
    rq.enqueue(23214);
    rq.enqueue(43);
    rq.enqueue(123);
    rq.enqueue(435);
    rq.enqueue(234);
    /*
    StdOut.println(rq.sample());
    StdOut.println(rq.dequeue());
    */
    Iterator<Integer> i = rq.iterator();
    while(!i.hasNext()){
        int s = i.next();
        StdOut.println(s);
    }
  }
}

可以返回最大值的栈结构

构造一个数据结构拥有栈的基本操作和一个可以返回最大值的方法

public static class MaxStack{
  private Node first;
  private Node max;

  private class Node{
    double item;
    Node next;
  }

  public MaxStack{
    first = null;
    max = null;
  }

  public double GetMax(){
    return max.item;
  }

  public void push(double item){
    if(item >= max.item){
      oldmax = max.item;
      max = new Node();
      max.item = item;
      max.next = oldmax;
    }
    oldfirst = first.item;
    first = new Node();
    first.item = item;
    first.next = oldfirst;
  }

  public double pop(){
    double temp = first.item;
    first = first.next;
    if(temp == max.item){
      max = max.next;
    }
    return temp;
  }
}

构造一个由两个栈组成的队列

如何构造一个由两个栈组成的队列

队列所需的操作

  • enqueue ：将一个元素加入到队列的末尾
  • dequeue ： 将一个元素从队列的顶部移除

栈的操作

  • pop : 将一个元素从栈顶移除
  • push ：将一个元素加入到栈顶

思路

  • 可以将队列分为两部分 inbox outbox(两个栈)
  • 当outbox为空的时候，将inbox的元素逐个pop并push到outbox中，此时outbox的栈顶为最先加入到inbox的元素

代码

public static class QueueWithTwoStakcs{
  private Stack<Item> inbox = new Stack<Item>;
  private Stack<Item> outbox = new Stack<Item>;

  public void enqueue(Item item){
    inbox.push(item);
  }

  public Item dequeue(){
    if(outbox.isEmpty()){
      while(!inbox.isEmpty){
        outbox.push(inbox.pop());
      }
    }
    return outbox.pop();
  }
}

JAVA迭代器

集合数据类型的基本操作之一就是，能够使用JAVA的foreach语句通过迭代遍历并处理集合中的每个元素。并且不依赖于集合数据类型的具体实现。
如：

Stack<String> collection = new Stack<String>();
...
for(String s : collection)
  StdOut.println(s);
...

等同于：

Iterator<String> i = collection.iterator();
while(i.hasNext()){
  String s = i.next();
  StdOut.println(s);
}

所以 任意可迭代的集合数据类型中 都需要实现：

  • 集合数据类型必须实现一个iterator（）方法并返回一个Iterator对象；
  • Iterator类中必须包含两个方法hasNext()（返回一个布尔值）和next()（返回集合的数据类型）

在JAVA中，使用接口机制来指定一个类所必须实现的方法。要使一个类可迭代，第一步就是在它的声明中加入implements Iterable,即:

public interface Iterable<Item>{
  Iterator<Item> iterator();
}

然后就可以在类中添加一个iterator方法，并返回一个迭代器Iterator。如：

public Iterator<Item> iterator(){
  return new ReverseArrayIterator();//迭代器的名字根据需求起
}

迭代器是什么？

迭代器是一个实现了hasNext()和next()方法的类的对象，由以下的接口定义：

public interface Iterator<Item>
{
  boolean hasNext();
  Item next();
}

然后就可以在一个类中的嵌套类实现这些方法:

private class ReverseArrayIterator implements Iterator<Item>
{
  private i = N;

  public boolean hasNext(){return i>0;}
  public Item next(){return a[--i]}
}

此处需要注意的是，嵌套类可以访问包含它的类的实例变量，在这里就是a[]和N。

import java.util.Iterator;

        this.n = n; //储存传入的参数n

        opened = new boolean[n*n+2];
        for(int i=1;i<n*n+1;i++){
            opened[i] = false;
        }
        opened[0] = true;
        opened[n*n+1] = true;
    }

    //将传入的二维定位（行、列）转化为一维的id
    private int index(int row, int col){  
        //当传入的行数为0时，返回第一个虚拟点位的id
        if(row==0){
            return 0;
        }
        //当传入行数为n+1时，返回最后一个点（虚拟点位）的id
        else if(row==n+1){
            return n*n+1;
        }
        //其余情况通过计算返回id
        else{
            int p = (row-1)*n+col;
            return p;            
        }                  
    }

    //使传入坐标点开放
    public void open(int row, int col){
        int p = index(row,col);
        opened[p] = true;
        //检查该点上下左右是否开放，如果开放则将两者连接
        boolean up = isOpen(row-1,col);
        boolean right = isOpen(row,col+1);
        boolean left = isOpen(row,col-1);
        boolean down = isOpen(row+1,col);
        if(up) uf.union(p,index(row-1,col));
        if(right) uf.union(p,index(row,col+1));    
        if(left) uf.union(p,index(row,col-1));
        if(down) uf.union(p,index(row+1,col));
    }
    //检查传入坐标的点是否开放
    public boolean isOpen(int row,int col){
        //当传入行数为0的时候，返回第一个虚拟点位的开放性
        if(row==0){
            return opened[0];
        }
        //当传入行数为n+1时，返回第二个虚拟点位（位于最后）的开放性
        else if(row==n+1){
            return opened[n*n+1];
        }
        //当行数超出边界时，返回fasle
        else if(row<0||col<=0||row>n+1||col>n){
            return false;
        }
        //其余情况通过计算p
        else{
            int p = index(row,col);
            return opened[p];            
        }
    }

    public boolean isFull(int row,int col){
        int p = index(row,col);
        return !opened[p];
    }
    //当前开放的点在系统中的占比
    public int numberOfOpenSites(){
        int openNum = 0;
        for(int i=1;i<n*n+1;i++){
            if(opened[i]){
                openNum++;
            }
        }
        return openNum;
    }
    //该系统是否渗透
    public boolean percolates(){
        return uf.connected(0,n*n+1);
    }
    //如果系统不渗透，持续随机打开系统中的点，直到系统渗透为止
    public void doPercolation(int n){
        while(!percolates()){
            int row = StdRandom.uniform(n)+1;
            int col = StdRandom.uniform(n)+1;
            open(row,col);
        }
    }

    public static void main(String[] args){
        int n = StdRandom.uniform(10);
        Percolation percolation = new Percolation(n);
        while(!percolation.percolates()){
            int row = StdRandom.uniform(n)+1;
            int col = StdRandom.uniform(n)+1;
            percolation.open(row,col);
        }
        StdOut.println("system is percolate");    
    }
}

import edu.princeton.cs.algs4.StdOut;
import edu.princeton.cs.algs4.StdStats;

public class PercolationStats {
    private double[] thresholdArray;//保存几次测试渗透系统阈值的数组
    private int t = 0;

    public PercolationStats(int n,int trials){
        t = trials;//测试次数
        thresholdArray = new double[t];
        //进行测试
        while(trials>0){
            Percolation percolation = new Percolation(n);
            percolation.doPercolation(n);
            double openNum = percolation.numberOfOpenSites();
            double sum = n*n;
            double threshold = openNum/sum;
            thresholdArray[trials-1] = threshold;
            trials--;
        }

    }

    //计算平均值
    public double mean(){
        return StdStats.mean(thresholdArray);
    }

    //计算标准差
    public double stddev(){
        return StdStats.stddev(thresholdArray);
    }
    //计算置信区间下限
    public double confidenceLo(){
        return mean()-(1.96*stddev()/Math.sqrt(t));
    }
    //计算置信区间上限
    public double confidenceHi(){
        return mean()+(1.96*stddev()/Math.sqrt(t));
    }
    //从命令行接收两个数字，（系统大小和测试次数），打印数据
    public static void main(String[] args){
        int n;
        int trials;
        n = Integer.parseInt(args[0]);
        trials = Integer.parseInt(args[1]);
        PercolationStats percolationStats = new PercolationStats(n,trials);
        StdOut.println("mean = "+percolationStats.mean());
        StdOut.println("stddev = "+percolationStats.stddev());
        StdOut.println("95% confidece interval = ["+percolationStats.confidenceLo()+","+percolationStats.confidenceHi()+"];");
    }   
}
