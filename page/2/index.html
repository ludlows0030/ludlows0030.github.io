  • 链表定义

    链表是一种递归的数据结构。它或者为null或者指向一个结点Node的引用。一个结点包括一个泛型的元素和指向另一条链表的应用

  • 结点的抽象数据类型

    private class Node{
  Item item;//泛型，我们希望用链表处理任意数据结构
  Node next;
}

  • 举例：构造一个含有元素to,be,or的链表

    Node first = new Node();
Node second = new Node();
Node third = new Node();

first.item = "to";
second.item = "be";
third.item = "or";

first.next = second;
second.next = third;
//third.next为默认的null

  • 从表头插入结点

    Node oldfirst = first;
first = new Node();
first.item = "not";
first.next = oldfirst;

  • 从表头删除结点

    first = first.next;

  • 在表尾插入结点

    Node oldlast = last;
last = new Node();
last.item = "not";
oldlast.next = last;

  • 链表的遍历

  //对每一个结点进行处理

Percolation.Java

import edu.princeton.cs.algs4.WeightedQuickUnionUF;
import edu.princeton.cs.algs4.StdRandom;
import edu.princeton.cs.algs4.StdOut;
public class Percolation {
    private boolean[] opened; //所有结点是否开放的布尔值数组
    private int n; // n*n大小的grid
    WeightedQuickUnionUF uf;

    public Percolation(int n){
        this.n = n; //储存传入的参数n
        uf = new WeightedQuickUnionUF(n*n+2); //利用并查集的类建立n*n+2大小的序列，第一位id为0，最后一位id为n*n-1

        //模拟开始时，出第一个点和最后一个点（两个虚拟点位）默认为开放，其余的都默认为关闭
        opened = new boolean[n*n+2];
        for(int i=1;i<n*n+1;i++){
            opened[i] = false;
        }
        opened[0] = true;
        opened[n*n+1] = true;
    }

    //将传入的二维定位（行、列）转化为一维的id
    private int index(int row, int col){  
        //当传入的行数为0时，返回第一个虚拟点位的id
        if(row==0){
            return 0;
        }
        //当传入行数为n+1时，返回最后一个点（虚拟点位）的id
        else if(row==n+1){
            return n*n+1;
        }
        //其余情况通过计算返回id
        else{
            int p = (row-1)*n+col;
            return p;            
        }                  
    }

    //使传入坐标点开放
    public void open(int row, int col){
        int p = index(row,col);
        opened[p] = true;
        //检查该点上下左右是否开放，如果开放则将两者连接
        boolean up = isOpen(row-1,col);
        boolean right = isOpen(row,col+1);
        boolean left = isOpen(row,col-1);
        boolean down = isOpen(row+1,col);
        if(up) uf.union(p,index(row-1,col));
        if(right) uf.union(p,index(row,col+1));    
        if(left) uf.union(p,index(row,col-1));
        if(down) uf.union(p,index(row+1,col));
    }
    //检查传入坐标的点是否开放
    public boolean isOpen(int row,int col){
        //当传入行数为0的时候，返回第一个虚拟点位的开放性
        if(row==0){
            return opened[0];
        }
        //当传入行数为n+1时，返回第二个虚拟点位（位于最后）的开放性
        else if(row==n+1){
            return opened[n*n+1];
        }
        //当行数超出边界时，返回fasle
        else if(row<0||col<=0||row>n+1||col>n){
            return false;
        }
        //其余情况通过计算p
        else{
            int p = index(row,col);
            return opened[p];            
        }
    }

    public boolean isFull(int row,int col){
        int p = index(row,col);
        return !opened[p];
    }
    //当前开放的点在系统中的占比
    public int numberOfOpenSites(){
        int openNum = 0;
        for(int i=1;i<n*n+1;i++){
            if(opened[i]){
                openNum++;
            }
        }
        return openNum;
    }
    //该系统是否渗透
    public boolean percolates(){
        return uf.connected(0,n*n+1);
    }
    //如果系统不渗透，持续随机打开系统中的点，直到系统渗透为止
    public void doPercolation(int n){
        while(!percolates()){
            int row = StdRandom.uniform(n)+1;
            int col = StdRandom.uniform(n)+1;
            open(row,col);
        }
    }

    public static void main(String[] args){
        int n = StdRandom.uniform(10);
        Percolation percolation = new Percolation(n);
        while(!percolation.percolates()){
            int row = StdRandom.uniform(n)+1;
            int col = StdRandom.uniform(n)+1;
            percolation.open(row,col);
        }
        StdOut.println("system is percolate");    
    }
}

PercolationStats.Java

import edu.princeton.cs.algs4.StdOut;
import edu.princeton.cs.algs4.StdStats;

public class PercolationStats {
    private double[] thresholdArray;//保存几次测试渗透系统阈值的数组
    private int t = 0;

    public PercolationStats(int n,int trials){
        t = trials;//测试次数
        thresholdArray = new double[t];
        //进行测试
        while(trials>0){
            Percolation percolation = new Percolation(n);
            percolation.doPercolation(n);
            double openNum = percolation.numberOfOpenSites();
            double sum = n*n;
            double threshold = openNum/sum;
            thresholdArray[trials-1] = threshold;
            trials--;
        }

    }

    //计算平均值
    public double mean(){
        return StdStats.mean(thresholdArray);
    }

    //计算标准差
    public double stddev(){
        return StdStats.stddev(thresholdArray);
    }
    //计算置信区间下限
    public double confidenceLo(){
        return mean()-(1.96*stddev()/Math.sqrt(t));
    }
    //计算置信区间上限
    public double confidenceHi(){
        return mean()+(1.96*stddev()/Math.sqrt(t));
    }
    //从命令行接收两个数字，（系统大小和测试次数），打印数据
    public static void main(String[] args){
        int n;
        int trials;
        n = Integer.parseInt(args[0]);
        trials = Integer.parseInt(args[1]);
        PercolationStats percolationStats = new PercolationStats(n,trials);
        StdOut.println("mean = "+percolationStats.mean());
        StdOut.println("stddev = "+percolationStats.stddev());
        StdOut.println("95% confidece interval = ["+percolationStats.confidenceLo()+","+percolationStats.confidenceHi()+"];");
    }   
}

public class Managers : MonoBehaviour{
  public static PlayerManager Player{get;private set;}
  public static InventoryManager Inventory{get;private set;}
  //声明

  private List<IGameManager> _startSequence;
  //用于保存Manager的List

  void Awake(){
    Player = GetComponent<PlayerManager>();
    Inventory = GetComponent<InventoryManager>();

    _startSequence = new List<IGameManager>();
    _startSequence.Add(Player);
    _startSequence.Add(Inventory);
    //将Manager添加到List中

    StartCoroutine(StartupManagers());
    //异步启动序列，初始化Manager
  }

  private IEnumerator StartupManagers(){
    foreach(IGameManager manager in _startSequence){
      manager.Startup();
    }

    yield return null;暂停一下

    int numModules = _startSequence.Count;
    //需初始化的模块总数
    int numReady = 0;
    //初始化完毕的模块数

    whilie(numReady<numModules){
      int lastReady = numReady;
      //记录当前初始化完成的模块数
      numReady = 0;

      foreach(IGameManager manager in _startSequence){
        if(manager.status == ManagerStatus.Started){
          //在所有模块初始化完毕之前，如果有模块初始化完毕，则numReady++
          numReady++
        }
      }

      if(numReady>lastReady){
        //如果初始化完毕的模块数大于当前所记录的初始化完毕的模块数（也就是lastReady的值），就更新
        Debug.Log("Process: "+numReady+"/"+numModules);
        yield retrun null;
      }
    }
    Debug.Log("All managers started up");
  }        
}

CRT显示器中电子枪是如何绘图的？

CRT显像管的成像原理是通过显像管内的电子枪，将光束射出，打在内层玻璃的三原色荧光粉层上，电子束使这些荧光粉层发光，形成图像。
一个像素点由红、绿、蓝三种基色组成，所以要由三个光束打在一个像素点上，并根据光束强弱的不同使红、绿、蓝三种基色的强弱不同，从而显现出各种颜色。由于一个像素点内三种基色靠的很近，人眼无法分辨，所以我们可以在显示器上看到图像。

什么是场消隐期

CRT显像系统通常采用2种扫描方式，即隔行扫描和逐行扫描。
以隔行扫描为例，隔行扫描是，一副图像分两次进行扫描，第一次扫描奇数行的光栅，第二次扫描偶数行的光栅，两场之间的回扫时间称为 场消隐期。
