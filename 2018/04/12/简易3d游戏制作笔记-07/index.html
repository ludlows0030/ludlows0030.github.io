简易3d游戏制作笔记-07

管理仓库数据和游戏状态（MVC设计模式）

基本设计思路

  • 使用PlayerManager维护玩家状态
  • 使用InventoryManager维护玩家的物品清单
  • 使用Managers来统一管理所有的Manager(如PlayerManager,InventoryManager)
  • 为了让Managers(主管理器)以一种一致的方式来引用其他模块，这些模块必须从一个基类继承共同的属性，将这个共同的基类定义为一个接口

具体代码设计

  • 基本的基类接口(IGameManager)

    //通常以I+接口名的格式来命名接口
public interface IGameManager{
  ManagerStatus status{get;}

  //各个Manager用来初始化的方法
  void Startup();
}

  • 实现枚举类型ManagerStatus

    public enum ManagerStatus{
  //管理器所有的可能状态
  Shutdown,
  Initializing,
  Started
}

  • 编写InventoryManager脚本

    //实现IGameManager接口
public class InventoryManager : MonoBehaviour, IGameManager{
  public ManagerStatus status {get;private set;}

  public void Startup(){
    //如果是长任务改为.Initializing
    stauts = ManagerStatus.Started;
  }
}

  • 编写PlayerManager脚本

    public class PlayerManager : MonoBehaviour, IGameManager{
  public ManagerStatus status{get;private set;}

  public int health {get;private set;}
  public int maxhealth {get;private set;}

  public void Startup(){
    health = 50;
    maxhealth = 100;
    初始化属性

    status = ManagerStatus.Started;
  }

  public void ChangeHealth(int value){
    health += value;
    if(health>maxhealth){
      health = maxhealth;
    }
    else if(health<0){
      health = 0;
    }
    //在控制台输出
    Debug.Log("Health: "+health+"/"+"maxhealth");
  }
}

  • 编写Managers脚本（主管理器）

    [RequireComponent(typeof(PlayerManager))]
[RequireComponent(typeof(InventoryManager))]

public class Managers : MonoBehaviour{
  public static PlayerManager Player{get;private set;}
  public static InventoryManager Inventory{get;private set;}
  //声明

  private List<IGameManager> _startSequence;
  //用于保存Manager的List

  void Awake(){
    Player = GetComponent<PlayerManager>();
    Inventory = GetComponent<InventoryManager>();

    _startSequence = new List<IGameManager>();
    _startSequence.Add(Player);
    _startSequence.Add(Inventory);
    //将Manager添加到List中

    StartCoroutine(StartupManagers());
    //异步启动序列，初始化Manager
  }

  private IEnumerator StartupManagers(){
    foreach(IGameManager manager in _startSequence){
      manager.Startup();
    }

    yield return null;暂停一下

    int numModules = _startSequence.Count;
    //需初始化的模块总数
    int numReady = 0;
    //初始化完毕的模块数

    whilie(numReady<numModules){
      int lastReady = numReady;
      //记录当前初始化完成的模块数
      numReady = 0;

      foreach(IGameManager manager in _startSequence){
        if(manager.status == ManagerStatus.Started){
          //在所有模块初始化完毕之前，如果有模块初始化完毕，则numReady++
          numReady++
        }
      }

      if(numReady>lastReady){
        //如果初始化完毕的模块数大于当前所记录的初始化完毕的模块数（也就是lastReady的值），就更新
        Debug.Log("Process: "+numReady+"/"+numModules);
        yield retrun null;
      }
    }
    Debug.Log("All managers started up");
  }        
}
