基于二分查找的有序数组实现的符号表

发表于 2018-05-27

基于二分查找的有序数组实现的符号表

符号表应该要实现哪些操作

  • 将一个键值对插入符号表
  • 在之后能够从符号表的所有键值对中按照键直接找到相应的值

符号表的定义

  • 符号表是一种存储键值对的数据结构，支持两种操作：插入（put），即将一组新的键值对存入表中；查找（get），即根据给定的键得到相应的值

符号表的关于重复键的规则

  • 每个键只对应一个值 并不允许存在重复的键
  • 当用例代码向表中存入的键值对和表中已有的键冲突时，新的值会替代旧的值

空键的情况

  • 键不能为空，使用空键会产生一个运行时的异常

空值的情况

  • 不允许有空值
  • 用get()方法返回空来测试给定的键是否存在于符号表
  • 我们可以将空值作为put()方法的第二个参数存入表中来实现删除

删除操作

  • 延时删除 ： 将键对应的值置为空，然后在某个时候删去所有值为空的键
  • 即时删除 ： 立刻从表中删除指定的键

排名和选择

  • 检验一个新的键是否插入合适位置的基本操作
  • rank() 找出小于指定键的键的数量
  • select() 找出排名为k的键
  • 通常有 i == rank(select(i)) 以及 key == select(rank（key))

实际情况中的能处理大量初级的符号表应具备的特性

  • 混合使用查找和插入的操作
  • 大量不同的键
  • 查找操作比插入操作多得多
  • 虽然不可预测，但查找和插入操作的使用模式并非随机

有序数组的二分查找

  • 核心是rank()方法，用于定位，确定位置后，再用put()和get()方法

具体实现

public class BinarySearchST<Key extends Comparable<key>, Value>
{
    //两个数组，一个存储键，一个存储值
    private Key[] keys;
    private Value[] vals;
    private int N;

    //构造函数，构造传入参数大小的数组
    public BinarySearchST(int capacity)
    {
        keys = (Key[]) new Comparable[capacity];
        vals = (Value[]) new Comparable[capacity];
    }

    //当表内存在键时，更新值，否则插入一个新的键值对
    public void put(Key key,Value val)
    {
        if(val = null) delete(key);return;
        int i = rank(key);
        if(i<N&&keys[i].compareTo(key) == 0)
        {
            vals[i] = val;
        }


        for(int j = N;j>i;j--)
        {
            keys[j] = keys[j-1];
            vals[j] = vals[j-1];
        }
        keys[i] = key;
        vals[i] = vals;
        N++;
    }

    //表为空时 返回null，key不在表内时返回null
    public Value get(Key key)
    {
        if(isEmpty())return null;
        int i = rank(key);
        if(i<N&&keys[i].compareTo(key)==0) return vals[i];
        else return null;
    }

    //删除键值对，并把其后键值对向前移
    public Key delete(Key key)
    {
        int i = rank(key);
        if(i<N&&keys[i].compareTo(key) == 0 )
        {
            vals[i] = null;
            for(int j = i;j<N;j++)
            {
                keys[j] = keys[j+1];
                vals[j] = vals[j+1];
            }
            N--;
            return key;
        }
        else return null;
    }

    //返回表内是否存在key键的布尔值
    public boolean contains(Key key)
    {
        int i = rank(key);
        return i<N&&keys[i].compareTo(key) == 0;
    }

    //表是否为空
    public boolean isEmpty()
    {
        return size() == 0;
    }

    //表的大小
    public int size()
    {
        return N;
    }

    //表中最小键
    public Key min()
    {
        return keys[0];
    }

    //表中最大键
    public Key max()
    {
        return keys[N-1];
    }

    //表中小于等于key键的最大键
    public Key floor(Key key)
    {
        int i = rank(key);
        return keys[i-1]
    }

    //表中大于等于key键的最小键
    public Key ceiling(Key key)
    {
        int i = rank(key);
        return keys[i];
    }

    //二分查找
    public int rank(Key key,int lo,int hi)
    {
        if(hi<lo) return lo;
        int mid = lo +(hi-lo)/2;
        int cmp = key.compareTo(keys[mid]);
        if(cmp<0)
            rank(key,lo,mid-1);
        else if(cmp>0)
            rank(key,mid+1,hi);
        else
            return mid;
    }

    //二分查找 非迭代
    public int rank(Key key)
    {
        int lo = 0,hi = N-1;
        while(lo<=hi)
        {
            int mid = lo+(hi-lo)/2;
            int cmp = key.compareTo(keys[mid]);
            if(cmp<0) hi = mid-1;
            else if (cmp>0) lo = mid+1;
            else return mid;
        }
    }

    //返回表中第k-1个key
    public Key select(int k)
    {
        return keys[k];
    }

    //删除最小值
    public void deleteMin()
    {
        delete(min());
    }

    //删除最大值
    public void deleteMax()
    {
        delete(max());
    }

    //返回指定区域的大小
    public int size(Key lo,Key hi)
    {
        if(hi.compareTo(lo)<0)
            return 0;
        else if(contains(hi))
            return rank(hi)-rank(lo)+1;
        else
            return rank(hi)-rank(lo);
    }

    //返回一个队列，存储指定区间内的键
    public Iterable<Key> keys(Key lo, Key hi)
    {
        Queue<Key> q = new Queue<Key>();
        for(int i = rank(lo);i<rank(hi);i++)
        {
            q.enqueue(keys[i]);
            if(contains(hi))
                q.enqueue(keys[rank(hi)]);
            return q;
        }
    }

    public Iterable<Key> keys()
    {
        return keys(min(),max());
    }
}
