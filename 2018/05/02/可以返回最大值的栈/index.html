可以返回最大值的栈

可以返回最大值的栈结构

构造一个数据结构拥有栈的基本操作和一个可以返回最大值的方法

public static class MaxStack{
  private Node first;
  private Node max;

  private class Node{
    double item;
    Node next;
  }

  public MaxStack{
    first = null;
    max = null;
  }

  public double GetMax(){
    return max.item;
  }

  public void push(double item){
    if(item >= max.item){
      oldmax = max.item;
      max = new Node();
      max.item = item;
      max.next = oldmax;
    }
    oldfirst = first.item;
    first = new Node();
    first.item = item;
    first.next = oldfirst;
  }

  public double pop(){
    double temp = first.item;
    first = first.next;
    if(temp == max.item){
      max = max.next;
    }
    return temp;
  }
}
